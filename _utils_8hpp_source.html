51  static std::vector<meshkernel::Point> ConvertGeometryListToPointVector(const GeometryList& geometryListIn)
52  {
53  std::vector<meshkernel::Point> result;
54  if (geometryListIn.num_coordinates == 0)
55  {
56  return result;
57  }
58  result.reserve(geometryListIn.num_coordinates);
59  result.emplace_back(geometryListIn.coordinates_x[0], geometryListIn.coordinates_y[0]);
60  // remove consecutive duplicated point leading to 0 edge length
61  for (auto i = 1; i < geometryListIn.num_coordinates; ++i)
62  {
63  if (meshkernel::IsEqual(geometryListIn.coordinates_x[i], result.back().x) &&
64  meshkernel::IsEqual(geometryListIn.coordinates_y[i], result.back().y))
65  {
66  continue;
67  }
68  result.emplace_back(geometryListIn.coordinates_x[i], geometryListIn.coordinates_y[i]);
69  }
70  return result;
71  }

77  static std::vector<std::vector<meshkernel::Point>> ConvertGeometryListToVectorOfPointVectors(GeometryList const& geometryListIn)
78  {
79  std::vector<std::vector<meshkernel::Point>> result;
80  std::vector<meshkernel::Point> chunk;
81  chunk.reserve(geometryListIn.num_coordinates);
82  for (auto i = 0; i < geometryListIn.num_coordinates; ++i)
83  {
84 
85  if (meshkernel::IsEqual(geometryListIn.coordinates_x[i], geometryListIn.geometry_separator))
86  {
87  result.emplace_back(chunk);
88  chunk.clear();
89  }
90  else
91  {
92  chunk.emplace_back(geometryListIn.coordinates_x[i], geometryListIn.coordinates_y[i]);
93  }
94  }
95 
96  if (!chunk.empty())
97  {
98  result.emplace_back(chunk);
99  }
100 
101  return result;
102  }

264  static void SetMesh2dApiData(std::shared_ptr<meshkernel::Mesh2D> mesh2d, Mesh2D& mesh2dApi)
265  {
266  mesh2d->ComputeEdgesCenters();
267  for (size_t n = 0; n < mesh2d->GetNumNodes(); n++)
268  {
269  mesh2dApi.node_x[n] = mesh2d->m_nodes[n].x;
270  mesh2dApi.node_y[n] = mesh2d->m_nodes[n].y;
271  }
272 
273  for (size_t edgeIndex = 0; edgeIndex < mesh2d->GetNumEdges(); edgeIndex++)
274  {
275  mesh2dApi.edge_x[edgeIndex] = mesh2d->m_edgesCenters[edgeIndex].x;
276  mesh2dApi.edge_y[edgeIndex] = mesh2d->m_edgesCenters[edgeIndex].y;
277  mesh2dApi.edge_nodes[edgeIndex * 2] = static_cast<int>(mesh2d->m_edges[edgeIndex].first);
278  mesh2dApi.edge_nodes[edgeIndex * 2 + 1] = static_cast<int>(mesh2d->m_edges[edgeIndex].second);
279 
280  const auto& firstEdgeFace = mesh2d->m_edgesFaces[edgeIndex][0];
281  mesh2dApi.edge_faces[edgeIndex * 2] = firstEdgeFace == meshkernel::constants::missing::uintValue ? -1 : static_cast<int>(firstEdgeFace);
282  const auto& secondEdgeFace = mesh2d->m_edgesFaces[edgeIndex][1];
283  mesh2dApi.edge_faces[edgeIndex * 2 + 1] = secondEdgeFace == meshkernel::constants::missing::uintValue ? -1 : static_cast<int>(secondEdgeFace);
284  }
285 
286  int faceIndex = 0;
287  for (size_t f = 0; f < mesh2d->GetNumFaces(); f++)
288  {
289  mesh2dApi.face_x[f] = mesh2d->m_facesMassCenters[f].x;
290  mesh2dApi.face_y[f] = mesh2d->m_facesMassCenters[f].y;
291  mesh2dApi.nodes_per_face[f] = static_cast<int>(mesh2d->m_facesNodes[f].size());
292  for (size_t n = 0; n < mesh2d->m_facesNodes[f].size(); ++n)
293  {
294  mesh2dApi.face_nodes[faceIndex] = static_cast<int>(mesh2d->m_facesNodes[f][n]);
295  mesh2dApi.face_edges[faceIndex] = static_cast<int>(mesh2d->m_facesEdges[f][n]);
296  faceIndex++;
297  }
298  }

342  static meshkernel::CurvilinearGrid CreateUniformCurvilinearGrid(const meshkernel::MakeGridParameters& makeGridParameters,
343  const GeometryList& geometryList,
344  const meshkernel::Projection& projection)
345  {
346  meshkernel::CurvilinearGridCreateUniform curvilinearGridCreateUniform(projection);
347 
348  auto polygonNodes = ConvertGeometryListToPointVector(geometryList);
349 
350  const auto polygon = std::make_shared<meshkernel::Polygons>(polygonNodes, projection);
351 
352  if (!polygon->IsEmpty())
353  {
354  return curvilinearGridCreateUniform.Compute(makeGridParameters.angle,
355  makeGridParameters.block_size_x,
356  makeGridParameters.block_size_y,
357  polygon,
358  0);
359  }
360  if (makeGridParameters.num_columns > 0 && makeGridParameters.num_rows > 0)
361  {
362  return curvilinearGridCreateUniform.Compute(makeGridParameters.num_columns,
363  makeGridParameters.num_rows,
364  makeGridParameters.origin_x,
365  makeGridParameters.origin_y,
366  makeGridParameters.angle,
367  makeGridParameters.block_size_x,
