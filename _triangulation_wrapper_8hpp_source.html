63  template <typename T>
64  void Compute(const std::vector<T>& inputNodes,
65  TriangulationOptions triangulationOption,
66  double averageTriangleArea,
67  size_t estimatedNumberOfTriangles)
68  {
69 
70  std::vector<double> xLocalPolygon(inputNodes.size());
71  std::vector<double> yLocalPolygon(inputNodes.size());
72  for (size_t i = 0; i < inputNodes.size(); ++i)
73  {
74  xLocalPolygon[i] = inputNodes[i].x;
75  yLocalPolygon[i] = inputNodes[i].y;
76  }
77 
78  m_numFaces = -1;
79  m_numEdges = 0;
80  m_numNodes = 0;
81 
82  int numInputNodes = static_cast<int>(inputNodes.size());
83  auto intTriangulationOption = static_cast<int>(triangulationOption);
84 
85  if (estimatedNumberOfTriangles == 0)
86  {
87  estimatedNumberOfTriangles = inputNodes.size() * 6 + 10;
88  }
89 
90  // If the number of estimated triangles is not sufficient, triangulation must be repeated
91  while (m_numFaces < 0)
92  {
93  m_numFaces = static_cast<int>(estimatedNumberOfTriangles);
94 
95  m_faceNodesFlat.resize(estimatedNumberOfTriangles * 3);
96  std::fill(m_faceNodesFlat.begin(), m_faceNodesFlat.end(), 0);
97 
98  m_edgeNodesFlat.resize(estimatedNumberOfTriangles * 2);
99  std::fill(m_edgeNodesFlat.begin(), m_edgeNodesFlat.end(), 0);
100 
101  m_faceEdgesFlat.resize(estimatedNumberOfTriangles * 3);
102  std::fill(m_faceEdgesFlat.begin(), m_faceEdgesFlat.end(), 0);
103 
104  m_xCoordFlat.resize(estimatedNumberOfTriangles * 3, constants::missing::doubleValue);
105  std::fill(m_xCoordFlat.begin(), m_xCoordFlat.end(), 0.0);
106 
107  m_yCoordFlat.resize(estimatedNumberOfTriangles * 3, constants::missing::doubleValue);
108  std::fill(m_yCoordFlat.begin(), m_yCoordFlat.end(), 0.0);
109 
110  Triangulation(&intTriangulationOption,
111  xLocalPolygon.data(),
112  yLocalPolygon.data(),
113  &numInputNodes,
114  m_faceNodesFlat.data(), // INDX
115  &m_numFaces,
116  m_edgeNodesFlat.data(), // EDGEINDX
117  &m_numEdges,
118  m_faceEdgesFlat.data(), // TRIEDGE
119  m_xCoordFlat.data(),
120  m_yCoordFlat.data(),
121  &m_numNodes,
122  &averageTriangleArea);
123  if (estimatedNumberOfTriangles > 0)
124  {
125  estimatedNumberOfTriangles = -m_numFaces;
126  }
127  }
128  }
