72  void Compute(const std::vector<T>& inputNodes,
73  TriangulationOptions triangulationOption,
74  double averageTriangleArea,
75  size_t estimatedNumberOfTriangles)
76  {
77 
78  std::vector<double> xLocalPolygon(inputNodes.size());
79  std::vector<double> yLocalPolygon(inputNodes.size());
80  for (size_t i = 0; i < inputNodes.size(); ++i)
81  {
82  xLocalPolygon[i] = inputNodes[i].x;
83  yLocalPolygon[i] = inputNodes[i].y;
84  }
85 
86  int numFaces = -1;
87  int numEdges = 0;
88  int numNodes = 0;
89 
90  int numInputNodes = triangulationOption == TriangulationOptions::TriangulatePointsAndGenerateFaces ? static_cast<int>(inputNodes.size()) : static_cast<int>(inputNodes.size() - 1);
91  auto intTriangulationOption = static_cast<int>(triangulationOption);
92 
93  std::vector<int> faceNodesFlat;
94  std::vector<int> edgeNodesFlat;
95  std::vector<int> faceEdgesFlat;
96  std::vector<double> xNodesFlat;
97  std::vector<double> yNodesFlat;
98 
99  if (estimatedNumberOfTriangles == 0)
100  {
101  estimatedNumberOfTriangles = inputNodes.size() * 6 + 10;
102  }
103 
104  // If the number of estimated triangles is not sufficient, triangulation must be repeated
105  while (numFaces < 0)
106  {
107  numFaces = static_cast<int>(estimatedNumberOfTriangles);
108  faceNodesFlat.resize(numFaces * 3);
109  edgeNodesFlat.resize(numFaces * 2);
110  faceEdgesFlat.resize(numFaces * 3);
111  xNodesFlat.resize(numFaces * 3, doubleMissingValue);
112  yNodesFlat.resize(numFaces * 3, doubleMissingValue);
113  Triangulation(&intTriangulationOption,
114  &xLocalPolygon[0],
115  &yLocalPolygon[0],
116  &numInputNodes,
117  &faceNodesFlat[0], // INDX
118  &numFaces,
119  &edgeNodesFlat[0], // EDGEINDX
120  &numEdges,
121  &faceEdgesFlat[0], // TRIEDGE
122  &xNodesFlat[0],
123  &yNodesFlat[0],
124  &numNodes,
125  &averageTriangleArea);
126  if (estimatedNumberOfTriangles > 0)
127  {
128  estimatedNumberOfTriangles = -numFaces;
129  }
130  }
131 
132  m_numFaces = numFaces <= 0 ? static_cast<size_t>(0) : static_cast<size_t>(numFaces);
133  m_numEdges = static_cast<size_t>(numEdges);
134  m_numNodes = static_cast<size_t>(numNodes);
135 
136  // Create nodes
137  m_nodes.resize(m_numNodes);
138  for (size_t i = 0; i < m_numNodes; ++i)
139  {
140  m_nodes[i] = {xNodesFlat[i], yNodesFlat[i]};
141  }
142 
143  // Create m_faceNodes
144  ResizeAndFill2DVector(m_faceNodes, m_numFaces, 3, true, sizetMissingValue);
145  ResizeAndFill2DVector(m_faceEdges, m_numFaces, 3, true, sizetMissingValue);
146  size_t faceCounter = 0;
147  for (size_t f = 0; f < m_numFaces; ++f)
148  {
149  m_faceNodes[f][0] = static_cast<size_t>(faceNodesFlat[faceCounter] - 1);
150  m_faceEdges[f][0] = static_cast<size_t>(faceEdgesFlat[faceCounter] - 1);
151  faceCounter++;
152  m_faceNodes[f][1] = static_cast<size_t>(faceNodesFlat[faceCounter] - 1);
153  m_faceEdges[f][1] = static_cast<size_t>(faceEdgesFlat[faceCounter] - 1);
154  faceCounter++;
155  m_faceNodes[f][2] = static_cast<size_t>(faceNodesFlat[faceCounter] - 1);
156  m_faceEdges[f][2] = static_cast<size_t>(faceEdgesFlat[faceCounter] - 1);
157  faceCounter++;
158  }
159 
160  // Create edges
161  if (m_numEdges == 0)
162  {
163  return;
164  }
165 
166  ResizeAndFill2DVector(m_edgeNodes, m_numEdges, 2, true, sizetMissingValue);
167  size_t edgeCounter = 0;
168  for (size_t e = 0; e < m_numEdges; ++e)
169  {
170  m_edgeNodes[e][0] = static_cast<size_t>(edgeNodesFlat[edgeCounter] - 1);
171  edgeCounter++;
172  m_edgeNodes[e][1] = static_cast<size_t>(edgeNodesFlat[edgeCounter] - 1);
173  edgeCounter++;
174  }
175 
176  ResizeAndFill2DVector(m_edgesFaces, m_numEdges, 2, true, sizetMissingValue);
177  edgeCounter = 0;
178  for (size_t f = 0; f < m_numFaces; ++f)
179  {
180 
181  for (size_t n = 0; n < numNodesInTriangle; ++n)
182  {
183  auto const edge = static_cast<size_t>(faceEdgesFlat[edgeCounter] - 1);
184  edgeCounter++;
185  // For each edge, the shared face index
186  if (m_edgesFaces[edge][0] == sizetMissingValue)
187  {
188  m_edgesFaces[edge][0] = f;
189  }
190  else
191  {
192  m_edgesFaces[edge][1] = f;
193  }
194  }
195  }
196  }
