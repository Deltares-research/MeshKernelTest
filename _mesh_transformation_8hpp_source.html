261  class MeshTransformation
262  {
263  public:
265  template <TransformationFunction Transformation>
266  static void Compute(Mesh& mesh, Transformation transformation)
267  {
268  if (mesh.m_projection != transformation.TransformationProjection())
269  {
270  throw MeshKernelError("Incorrect mesh coordinate system, expecting '{}', found '{}'",
271  ProjectionToString(transformation.TransformationProjection()), ProjectionToString(mesh.m_projection));
272  }
273 
274 #pragma omp parallel for
275  for (int i = 0; i < static_cast<int>(mesh.GetNumNodes()); ++i)
276  {
277  if (mesh.m_nodes[i].IsValid())
278  {
279  mesh.m_nodes[i] = transformation(mesh.m_nodes[i]);
280  }
281  }
282 
283  mesh.Administrate();
284  }
285  };
