181  template <typename F>
182  [[nodiscard]] double FindFunctionRootWithGoldenSectionSearch(F func, double min, double max)
183  {
184  // golden distance factors
185  const double c = 0.38196602;
186  const double r = 0.61803399;
187  const double tolerance = 1e-5;
188 
189  double left = min;
190  double middle = (min + max) * 0.5;
191  double right = max;
192 
193  double x0 = left;
194  double x1 = middle - c * (middle - left);
195  double x2 = middle;
196  double x3 = right;
197  if (std::abs(right - middle) > std::abs(middle - left))
198  {
199  x1 = middle;
200  x2 = middle + c * (right - left);
201  }
202 
203  double f1 = func(x1);
204  double f2 = func(x2);
205 
206  while (std::abs(x3 - x0) > tolerance * std::max(std::abs(x1) + std::abs(x2), 1e-8)) // tolerance * std::max(std::abs(x1) + std::abs(x2), 1e-10)
207  {
208  if (f2 < f1)
209  {
210  x0 = x1;
211  x1 = x2;
212  x2 = r * x1 + c * x3;
213 
214  f1 = f2;
215  f2 = func(x2);
216  }
217  else
218  {
219  x3 = x2;
220  x2 = x1;
221  x1 = r * x2 + c * x0;
222 
223  f2 = f1;
224  f1 = func(x1);
225  }
226  }
227 
228  return f1 < f2 ? x1 : x2;
229  }

440  template <typename T>
441  [[nodiscard]] T ComputePointOnSplineAtAdimensionalDistance(const std::vector<T>& coordinates,
442  const std::vector<T>& coordinatesDerivatives,
443  double pointAdimensionalCoordinate)
444  {
445  T pointCoordinate{};
446  if (pointAdimensionalCoordinate < 0)
447  {
448  return pointCoordinate;
449  }
450 
451  const double eps = 1e-5;
452  const double splFac = 1.0;
453  const auto intCoordinate = static_cast<double>(std::floor(pointAdimensionalCoordinate));
454  if (pointAdimensionalCoordinate - intCoordinate < eps)
455  {
456  return pointCoordinate = coordinates[intCoordinate];
457  }
458 
459  const size_t low = intCoordinate;
460  const size_t high = low + 1;
461  const double a = high - pointAdimensionalCoordinate;
462  const double b = pointAdimensionalCoordinate - low;
463 
464  pointCoordinate = coordinates[low] * a + coordinates[high] * b +
465  (coordinatesDerivatives[low] * (pow(a, 3) - a) + coordinatesDerivatives[high] * (pow(b, 3) - b)) / 6.0 * splFac;
466 
467  return pointCoordinate;
468  }

566  [[nodiscard]] std::tuple<Point, Point> GetBoundingBox(const std::vector<T>& points)
567  {

569  double maxx = std::numeric_limits<double>::lowest();
570  double miny = std::numeric_limits<double>::max();
571  double maxy = std::numeric_limits<double>::lowest();

574  {
575  if (point.IsValid())

577  minx = std::min(minx, point.x);
578  maxx = std::max(maxx, point.x);
579  miny = std::min(miny, point.y);
