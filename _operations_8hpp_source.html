138  template <typename F>
139  [[nodiscard]] double FindFunctionRootWithGoldenSectionSearch(F func, double min, double max)
140  {
141  //golden distance factors
142  const double c = 0.38196602;
143  const double r = 0.61803399;
144  const double tolerance = 0.00001;
145 
146  double left = min;
147  double middle = (min + max) * 0.5;
148  double right = max;
149 
150  double x0 = left;
151  double x1 = middle - c * (middle - left);
152  double x2 = middle;
153  double x3 = right;
154  if (std::abs(right - middle) > std::abs(middle - left))
155  {
156  x1 = middle;
157  x2 = middle + c * (middle - left);
158  }
159 
160  double f1 = func(x1);
161  double f2 = func(x2);
162 
163  while (std::abs(x3 - x0) > tolerance * std::max(std::abs(x1) + std::abs(x2), 1e-8))
164  {
165  if (f2 < f1)
166  {
167  x0 = x1;
168  x1 = x2;
169  x2 = r * x1 + c * x3;
170 
171  f1 = f2;
172  f2 = func(x2);
173  }
174  else
175  {
176  x3 = x2;
177  x2 = x1;
178  x1 = r * x2 + c * x0;
179 
180  f2 = f1;
181  f1 = func(x1);
182  }
183  }
184 
185  return f1 < f2 ? x1 : x2;
186  }

410  template <typename T>
411  [[nodiscard]] bool InterpolateSplinePoint(const std::vector<T>& coordinates,
412  const std::vector<T>& coordinatesDerivatives,
413  double pointAdimensionalCoordinate,
414  T& pointCoordinate)
415  {
416  if (pointAdimensionalCoordinate < 0)
417  {
418  return false;
419  }
420 
421  const double eps = 1e-5;
422  const double splFac = 1.0;
423  const auto intCoordinate = static_cast<size_t>(std::floor(pointAdimensionalCoordinate));
424  if (pointAdimensionalCoordinate - intCoordinate < eps)
425  {
426  pointCoordinate = coordinates[intCoordinate];
427  return true;
428  }
429 
430  size_t low = intCoordinate;
431  size_t high = low + 1;
432  double a = high - pointAdimensionalCoordinate;
433  double b = pointAdimensionalCoordinate - low;
434 
435  pointCoordinate = coordinates[low] * a + coordinates[high] * b +
436  (coordinatesDerivatives[low] * (pow(a, 3) - a) + coordinatesDerivatives[high] * (pow(b, 3) - b)) / 6.0 * splFac;
437 
438  return true;
439  }

529  {
530  double minx = std::numeric_limits<double>::max();
531  double maxx = std::numeric_limits<double>::lowest();
532  double miny = std::numeric_limits<double>::max();
533  double maxy = std::numeric_limits<double>::lowest();
