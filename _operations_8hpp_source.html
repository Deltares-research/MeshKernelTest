139  template <typename F>
140  [[nodiscard]] double FindFunctionRootWithGoldenSectionSearch(F func, double min, double max)
141  {
142  //golden distance factors
143  const double c = 0.38196602;
144  const double r = 0.61803399;
145  const double tolerance = 0.00001;
146 
147  double left = min;
148  double middle = (min + max) * 0.5;
149  double right = max;
150 
151  double x0 = left;
152  double x1 = middle - c * (middle - left);
153  double x2 = middle;
154  double x3 = right;
155  if (std::abs(right - middle) > std::abs(middle - left))
156  {
157  x1 = middle;
158  x2 = middle + c * (middle - left);
159  }
160 
161  double f1 = func(x1);
162  double f2 = func(x2);
163 
164  while (std::abs(x3 - x0) > tolerance * std::max(std::abs(x1) + std::abs(x2), 1e-8))
165  {
166  if (f2 < f1)
167  {
168  x0 = x1;
169  x1 = x2;
170  x2 = r * x1 + c * x3;
171 
172  f1 = f2;
173  f2 = func(x2);
174  }
175  else
176  {
177  x3 = x2;
178  x2 = x1;
179  x1 = r * x2 + c * x0;
180 
181  f2 = f1;
182  f1 = func(x1);
183  }
184  }
185 
186  return f1 < f2 ? x1 : x2;
187  }

347  template <typename T>
348  [[nodiscard]] bool InterpolateSplinePoint(const std::vector<T>& coordinates,
349  const std::vector<T>& coordinatesDerivatives,
350  double pointAdimensionalCoordinate,
351  T& pointCoordinate)
352  {
353  if (pointAdimensionalCoordinate < 0)
354  {
355  return false;
356  }
357 
358  const double eps = 1e-5;
359  const double splFac = 1.0;
360  const auto intCoordinate = int(std::floor(pointAdimensionalCoordinate));
361  if (pointAdimensionalCoordinate - intCoordinate < eps)
362  {
363  pointCoordinate = coordinates[intCoordinate];
364  return true;
365  }
366 
367  int low = intCoordinate;
368  int high = low + 1;
369  double a = high - pointAdimensionalCoordinate;
370  double b = pointAdimensionalCoordinate - low;
371 
372  pointCoordinate = coordinates[low] * a + coordinates[high] * b +
373  (coordinatesDerivatives[low] * (pow(a, 3) - a) + coordinatesDerivatives[high] * (pow(b, 3) - b)) / 6.0 * splFac;
374 
375  return true;
376  }

424  template <typename T>
425  void GetBoundingBox(const std::vector<T>& values, Point& lowerLeft, Point& upperRight)
426  {
427 
428  double minx = std::numeric_limits<double>::max();
429  double maxx = std::numeric_limits<double>::lowest();
430  double miny = std::numeric_limits<double>::max();
431  double maxy = std::numeric_limits<double>::lowest();
432  for (int n = 0; n < values.size(); n++)
433  {
434  bool isInvalid = IsEqual(values[n].x, doubleMissingValue) ||
435  IsEqual(values[n].y, doubleMissingValue);
436 
437  if (isInvalid)
438  {
439  continue;
440  }
441 
442  minx = std::min(minx, values[n].x);
443  maxx = std::max(maxx, values[n].x);
444  miny = std::min(miny, values[n].y);
445  maxy = std::max(maxy, values[n].y);
446  }
447 
448  lowerLeft = {minx, miny};
449  upperRight = {maxx, maxy};
450  }
