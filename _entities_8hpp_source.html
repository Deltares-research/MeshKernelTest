52  // template <std::floating_point T> // prefer this in this c++20
53  template <typename T, typename std::enable_if<std::is_floating_point<T>::value>::type* = nullptr>
54  static bool IsEqual(const T value, T ref_value, T eps_mutilpier = 10.0)
55  {
56  if (value == ref_value)
57  {
58  return true;
59  }
60  else
61  {
62  const T abs_diff = std::abs(value - ref_value);
63  const T abs_value = std::abs(value);
64  const T abs_ref_value = std::abs(ref_value);
65  static const T tol = eps_mutilpier * std::numeric_limits<T>::epsilon();
66  return abs_diff < tol * std::min(abs_value, abs_ref_value);
67  }
68  }

306  {
