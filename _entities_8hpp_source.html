52  template <std::floating_point T>
53  static bool IsEqual(const T value, T ref_value, T relative_tol = 10.0 * std::numeric_limits<T>::epsilon())
54  {
55 
56  if (value == ref_value)
57  {
58  return true;
59  }
60 
61  const T abs_diff = std::abs(value - ref_value);
62  const T abs_value = std::abs(value);
63  const T abs_ref_value = std::abs(ref_value);
64 
65  return abs_diff < relative_tol * std::min(abs_value, abs_ref_value);
66  }

310  static std::vector<std::vector<UInt>> ConvertToFaceNodesVector(int num_faces,
311  const int* const face_nodes,
312  const int* const nodes_per_face)
313  {
314  std::vector<std::vector<UInt>> result;
315  result.reserve(num_faces);
316 
317  std::vector<UInt> nodes;
318  UInt index = 0;
319  for (auto f = 0; f < num_faces; f++)
320  {
321  nodes.clear();
322  for (auto n = 0; n < nodes_per_face[f]; n++)
323  {
324  nodes.emplace_back(static_cast<UInt>(face_nodes[index]));
325  index++;
326  }
327  result.emplace_back(nodes);
328  }
329  return result;
